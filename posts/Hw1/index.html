Creating a Database

From the file temps.csv, station-metadata.csv, fips-10-4-to-iso-country-codes.csv, we can build a database with three such tables. From the class we did a little data wrangling to split the month variable for temps csv, and I am going to do the same.

I will first work on the temperature table. By using chunksize argument to make temps.csv into managable chunks and use prepare_df function to generate month column based on original “valuex” columns and divide temperature by 100 to get the Celsius for temperature.

conn = sqlite3.connect("temps.db") # create a database in current directory called temps.db
df_iter = pd.read_csv("temps.csv", chunksize = 100000)
def prepare_df(df):
    """
    Using prepare_df to change all the valuex to a month column using stack and setindex function. 
    Finnally change the temperature to celsius degree.
    """
    df = df.set_index(keys=["ID", "Year"])
    df = df.stack()
    df = df.reset_index()
    df = df.rename(columns = {"level_2"  : "Month" , 0 : "Temp"})
    df["Month"] = df["Month"].str[5:].astype(int)
    df["Temp"]  = df["Temp"] / 100
    return(df)
for i, df in enumerate(df_iter):
    df = prepare_df(df)
    df.to_sql("temperatures", conn, if_exists = "replace" if i == 0 else "append", index = False)

2. Write a Query Function

Import query_climate_database from climate_database.py which I used sql manipulation to produce panda dataframe needed for this task. Since the first two letters of country id in the station table is coresponding to FIPS 10-4 columns for countries table, we can use JOIN and SUBSTRING function to help us to display the full country name.

from climate_database import query_climate_database
import inspect
print(inspect.getsource(query_climate_database))  # import query function and inspect it
def query_climate_database(db_file, country, year_begin, year_end, month):
    conn = sqlite3.connect(db_file)
    cmd = \
        '''SELECT S.NAME , S.LATITUDE, S.LONGITUDE, C.Name, T.Year, T.Month, T.Temp
        FROM stations S
        LEFT JOIN temperatures T on S.id = T.id
        LEFT JOIN countries C on C.[FIPS 10-4]= SUBSTRING(S.id, 1, 2)
        WHERE C.NAME = {country} AND T.year >= {year_begin} AND T.year <= {year_end} AND T.month = {month}
        ''' .format(country = repr(country), year_begin = year_begin, year_end = year_end, month = month)
    result_df = pd.read_sql_query(cmd, conn)
    conn.close()
    result_df = result_df.rename(columns={'Name' : 'Country'})
    return result_df

3. Write a Geographic Scatter Function for Yearly Temperature Increases

With LinearRegression from sklearn, we can get a coeficient of temperature function based on the data.

from sklearn.linear_model import LinearRegression

def coef(data_group):
    x = data_group[["Year"]]  # 2 brackets because X should be a df
    y = data_group["Temp"]    # 1 bracket because y should be a series
    LR = LinearRegression()
    LR.fit(x, y)
    return round(LR.coef_[0], 4)  # round it to force decimal 
def temperature_coefficient_plot(db_file, country, year_begin, year_end, month, min_obs, **kwargs):
    #  using query_climate_database to get our temperature dataframe
    df = query_climate_database(db_file, country, year_begin, year_end, month)
    #  using a simple for loop to check how many observance made by each unique stations and drop based on min_obs
    obs = []
    for station in df['NAME'].unique():
        obs.append([station, df.isin([station]).sum()['NAME']])
    obs = pd.DataFrame(obs)
    df = pd.merge(df, obs, left_on = "NAME", right_on = 0).drop(columns = [0])
    df = df[df[1]>min_obs].drop(columns = [1])
    #  using coef to take the first coefficient of linear regression for our data and round it to the forth decimal place. 
    Estimated = pd.DataFrame(df.groupby('NAME').apply(coef))
    #  Then merge it into the original dataframe
    df = pd.merge(df, Estimated, on = 'NAME')
    df = df.rename({0 : 'Estimated Yearly Increase (°C)'}, axis=1)
    #  draw the graph, using calendar.month_name to convert month number to month name. 
    title = f'Estimate of yearly increase in temperatures in {calendar.month_name[month]} for stations in {country}, year {year_begin}-{year_end}'
    fig = px.scatter_mapbox(df,
                        lat = "LATITUDE",
                        lon = "LONGITUDE",
                        hover_name = "NAME",
                        color = "Estimated Yearly Increase (°C)",
                        title = title,
                        **kwargs)
    return fig
color_map = px.colors.diverging.RdGy_r

(b) Creating the Figures

This is a plot of average temperature in a country for multiple years for different months with px.bar function.

def temp_in_a_year(db_file, country, year1, year2, month_begin, month_end, **kwargs):
    """
    With our query function we can get any countries year month data from the database. 
    Using year as facet col to show multipe year's plot in one graph
    """
    df = query(db_file, country, year1, year2, month_begin, month_end)
    title = f'Average temperature from {calendar.month_name[month_begin]} to {calendar.month_name[month_end]} for stations in {country}, year {year1} to {year2}'
    new_df = pd.DataFrame(df.groupby(['Year', 'Month'])['Temp'].apply(np.average)).reset_index()
    fig = px.bar(new_df, x="Month", y ="Temp", title = title, facet_col = "Year", facet_col_wrap=4, **kwargs)
    return fig
fig = temp_in_a_year("temps.db", "India", 2002, 2012, 1, 12)
fig.show()
fig = temp_in_a_year("temps.db", "United States", 1999, 1, 12)

We can also show a graph of temperature of same month over the years for a specific country with px.line figures which is more suited for over the year changes.

def temp_in_a_month(db_file, country, year_begin, year_end, month, **kwargs):
    """
    Using query function to take a single months data for multiple years
    Using px.line to graph this
    """
    df = query(db_file, country, year_begin, year_end, month, month)
    title = f'Average temperature in {calendar.month_name[month]} for stations in {country}, year {year_begin} to {year_end}'
    new_df = pd.DataFrame(df.groupby('Year')['Temp'].apply(np.average))
    fig = px.line(new_df, title = title, **kwargs)
    return fig
fig = temp_in_a_month("temps.db", "India", 1999, 2010, 12)
fig.show()
fig = temp_in_a_month("temps.db", "United States", 1980, 2010, 12)
